工作目录
/etc/gitlab-runner/certs/client

# 生成 CA 私钥
openssl genrsa -out ca-key.pem 4096

# 生成 CA 证书
openssl req -new -x509 -days 3650 -key ca-key.pem -sha256 -out ca.pem -subj "/CN=DockerCA"

# 生成服务器私钥
openssl genrsa -out server-key.pem 4096

# 生成服务器 CSR (替换 YOUR_SERVER_HOST 为实际服务器主机名或IP)
openssl req -subj "/CN=YOUR_SERVER_HOST" -sha256 -new -key server-key.pem -out server.csr

# 创建扩展文件
echo "subjectAltName = IP:YOUR_SERVER_IP, DNS:YOUR_SERVER_HOST" > extfile.cnf
echo "extendedKeyUsage = serverAuth" >> extfile.cnf

# 用 CA 签名服务器证书
openssl x509 -req -days 3650 -sha256 -in server.csr -CA ca.pem -CAkey ca-key.pem -CAcreateserial -out server-cert.pem -extfile extfile.cnf

# 生成客户端私钥
openssl genrsa -out client-key.pem 4096

# 生成客户端 CSR
openssl req -subj "/CN=client" -new -key client-key.pem -out client.csr

# 创建客户端扩展文件
echo "extendedKeyUsage = clientAuth" > extfile-client.cnf

# 用 CA 签名客户端证书
openssl x509 -req -days 3650 -sha256 -in client.csr -CA ca.pem -CAkey ca-key.pem -CAcreateserial -out client-cert.pem -extfile extfile-client.cnf


# 清理临时文件
rm -v server.csr client.csr extfile.cnf extfile-client.cnf ca.srl

# 设置权限
chmod 0400 ca-key.pem server-key.pem client-key.pem
chmod 0444 ca.pem server-cert.pem client-cert.pem
